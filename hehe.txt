  images.forEach(img => {
    img.addEventListener('click', () => {
      const src = img.getAttribute('data-popup');
      popupImg.src = src;
      popup.style.display = 'flex';
      loveText.innerHTML = loveMessages[src] || "";
    });
  });

  backBtn.addEventListener('click', () => {
    popup.style.display = 'none';
    popupImg.src = "";
    loveText.innerHTML = "";
  });

  feelBtn.addEventListener('click', () => {
    phoneContainer.innerHTML = `
      <img src="https://i.imgur.com/PxOZFiA.png" style="width:100%; height:100%; object-fit:cover;">
      <button id="backBtn">back</button>
    `;
    const newBackBtn = document.getElementById('backBtn');
    newBackBtn.addEventListener('click', () => {
      phoneContainer.innerHTML = originalHTML;
      attachOverlayListeners();
    });
  });

  function attachOverlayListeners() {
    const newImages = document.querySelectorAll('.overlay-img');
    newImages.forEach(img => {
      img.addEventListener('click', () => {
        const src = img.getAttribute('data-popup');
        popupImg.src = src;
        popup.style.display = 'flex';
        loveText.innerHTML = loveMessages[src] || "";
      });
    });

    document.getElementById('feelBtn').addEventListener('click', () => {
      phoneContainer.innerHTML = `
        <img src="https://i.imgur.com/PxOZFiA.png" style="width:100%; height:100%; object-fit:cover;">
        <button id="backBtn">back</button>
      `;
      const newBackBtn = document.getElementById('backBtn');
      newBackBtn.addEventListener('click', () => {
        phoneContainer.innerHTML = originalHTML;
        attachOverlayListeners();
      });
    });
  }
